const eventDetailsEl = document.getElementById('event-details');

const eventPlaceholderEl = document.getElementById('event-placeholder');

const eventContentEl = document.getElementById('event-content');

const eventTitleEl = document.getElementById('event-title');

const eventTimeEl = document.getElementById('event-time');

const eventDescriptionEl = document.getElementById('event-description');

const eventRegisterBtnEl = document.getElementById('event-register-btn');


let currentDate = new Date();

let selectedDateStr = null;


function renderCalendar(year, month) {

calendarGridEl.innerHTML = '';

const monthNames = translations[currentLang].monthNames;

const dayNames = translations[currentLang].dayNames;


monthYearEl.textContent = `${monthNames[month]} ${year}`;


const firstDayOfMonth = new Date(year, month, 1).getDay();

const daysInMonth = new Date(year, month + 1, 0).getDate();


// Add day names

dayNames.forEach(day => {

const dayEl = document.createElement('div');

dayEl.className = 'font-bold text-sm text-gray-500';

dayEl.textContent = day;

calendarGridEl.appendChild(dayEl);

});


// Add empty cells for padding days

for (let i = 0; i < firstDayOfMonth; i++) {

calendarGridEl.appendChild(document.createElement('div'));

}


// Add day cells

for (let day = 1; day <= daysInMonth; day++) {

const dayEl = document.createElement('div');

dayEl.textContent = day;

dayEl.className = 'p-2 calendar-day';

const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;


if (events[dateStr]) {

dayEl.classList.add('has-event');

dayEl.dataset.date = dateStr;

dayEl.addEventListener('click', () => showEventDetails(dateStr));

}

if (dateStr === selectedDateStr) {

dayEl.classList.add('selected');

}


calendarGridEl.appendChild(dayEl);

}

}


function showEventDetails(dateStr) {

// Remove previous selection

const prevSelected = document.querySelector('.calendar-day.selected');

if (prevSelected) {

prevSelected.classList.remove('selected');

}


// Add new selection

const newSelected = document.querySelector(`.calendar-day[data-date="${dateStr}"]`);

if (newSelected) {

newSelected.classList.add('selected');

}

selectedDateStr = dateStr;

const eventData = events[dateStr][currentLang];


if (eventData) {

eventPlaceholderEl.classList.add('hidden');

eventContentEl.classList.remove('hidden');


eventTitleEl.textContent = eventData.title;

eventTimeEl.textContent = eventData.time;

eventDescriptionEl.textContent = eventData.description;

// IMPORTANT: Change the email address and subject line

const email = "registrace@nechmerust.org";

const subject = `Registrace na akci: ${eventData.title}`;

const body = `Dobrý den,\nmám zájem o registraci na akci "${eventData.title}" konanou dne ${dateStr}.\n\nJméno a příjmení:\nPočet osob:\n\nDěkuji.`;

eventRegisterBtnEl.href = `mailto:${email}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;


}

}

function updateEventDetailsForLangChange() {

if (selectedDateStr && events[selectedDateStr]) {

showEventDetails(selectedDateStr);

} else {

// No event was selected, just hide the details

eventPlaceholderEl.classList.remove('hidden');

eventContentEl.classList.add('hidden');

}

}



prevMonthBtn.addEventListener('click', () => {

currentDate.setMonth(currentDate.getMonth() - 1);

renderCalendar(currentDate.getFullYear(), currentDate.getMonth());

});


nextMonthBtn.addEventListener('click', () => {

currentDate.setMonth(currentDate.getMonth() + 1);

renderCalendar(currentDate.getFullYear(), currentDate.getMonth());

});


// --- MOBILE MENU ---

const mobileMenuButton = document.getElementById('mobile-menu-button');

const mobileMenu = document.getElementById('mobile-menu');

mobileMenuButton.addEventListener('click', () => {

mobileMenu.classList.toggle('hidden');

});

// Close mobile menu when a link is clicked

document.querySelectorAll('#mobile-menu a').forEach(link => {

link.addEventListener('click', () => {

mobileMenu.classList.add('hidden');

});

});


// --- FADE-IN ANIMATION ON SCROLL ---

const sections = document.querySelectorAll('.fade-in-section');

const observer = new IntersectionObserver((entries) => {

entries.forEach(entry => {

if (entry.isIntersecting) {

entry.target.classList.add('is-visible');

}

});

}, { threshold: 0.1 });

sections.forEach(section => observer.observe(section));



// --- INITIALIZE ON PAGE LOAD ---

window.addEventListener('load', () => {

document.getElementById('footer-year').textContent = new Date().getFullYear();

setLanguage('cs'); // Set initial language

renderCalendar(currentDate.getFullYear(), currentDate.getMonth());

});


</script>
